const curring = function (func) {
    //创建闭包
    const args = []
    //返回一个函数来循环收集参数，如果参数不为0，则递归本身
    return function result(...rest) {
        if (rest.length === 0) {
            //如果参数为0，则是调用，那么就把所有参数返还给回调函数，让回调函数去处理
            return func(...args)
        } else {
            //如果没有调用，则收集传入的闭包args中
            args.push(...rest)
            return result
        }
    }
}

function sort(...arg) {
    return arg.sort((x, y) => x - y)
}
curring(sort)(7, 8)(1)(5, 9)()
